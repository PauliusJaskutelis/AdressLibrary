CREATE PROCEDURE [dbo].[UpdateContact]
@id int,
@name varchar(250),
@number varchar(12),
@birthDate DATE
AS
BEGIN
	UPDATE adresai
	SET pilnas_vardas = @name,
		telefono_numeris = @number,
		gimimo_data = @birthDate
	WHERE id = @id
END
